## Usage

Import the component:

```tsx
import { AchievementBadge } from "@/components/ui/achievement-badge"
```

Basic usage:

```tsx
<AchievementBadge
  achievement={{
    id: "1",
    name: "Early Bird",
    trigger: "api",
    description: "Complete a task before 9am",
    achievedAt: new Date().toISOString(),
  }}
/>
```

With Trophy API data:

```tsx
const achievements = await trophy.achievements.list({ userId: "user_123" })

{achievements.map((achievement) => (
  <AchievementBadge key={achievement.id} achievement={achievement} />
))}
```

## Examples

### Locked Achievement

Achievements that haven't been unlocked yet.

```tsx
<AchievementBadge
  achievement={{
    id: "2",
    name: "First Steps",
    trigger: "api",
    description: "Complete your first task",
    achievedAt: null,
  }}
/>
```

### With Progress

Show progress toward metric-based achievements.

```tsx
<AchievementBadge
  achievement={achievement}
  showProgress
  currentValue={75}
/>
```

## API Reference

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `achievement` | `UserAchievement` | Required | Achievement data |
| `showDescription` | `boolean` | `true` | Show description below name |
| `showProgress` | `boolean` | `false` | Show progress for metric achievements |
| `currentValue` | `number` | - | Current value for metric progress |
| `lockedStyle` | `"grayscale" \| "silhouette" \| "hidden"` | `"grayscale"` | Style for locked achievements |
| `showLockIcon` | `boolean` | `true` | Show lock icon on locked achievements |

### Types

```ts
interface Achievement {
  id: string
  name: string
  trigger: "metric" | "api" | "streak"
  description?: string | null
  badgeUrl?: string | null
  metricValue?: number
}

interface UserAchievement extends Achievement {
  achievedAt: string | null
}
```
