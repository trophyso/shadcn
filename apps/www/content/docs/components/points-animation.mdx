## Usage

Import the component:

```tsx
import { PointsAnimation } from "@/components/ui/points-animation"
```

Basic usage:

```tsx
<PointsAnimation value={100} prefix="+" suffix=" XP" />
```

## Examples

### Animation Variants

```tsx
<PointsAnimation value={100} variant="float" />
<PointsAnimation value={100} variant="pop" />
<PointsAnimation value={100} variant="slide" />
```

### Colors

```tsx
<PointsAnimation value={100} color="green" />
<PointsAnimation value={100} color="gold" />
<PointsAnimation value={100} color="blue" />
```

### Using the Container

For managing multiple animations:

```tsx
import { 
  PointsAnimationContainer, 
  usePointsAnimation 
} from "@/components/ui/points-animation"

function GameUI() {
  const { items, addPoints, removeItem } = usePointsAnimation()

  return (
    <>
      <button onClick={() => addPoints(50)}>
        Earn Points
      </button>
      <PointsAnimationContainer 
        items={items} 
        onItemComplete={removeItem} 
      />
    </>
  )
}
```

## API Reference

### PointsAnimation Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `number` | Required | Points value to display |
| `prefix` | `string` | `""` | Prefix (e.g., "+") |
| `suffix` | `string` | `""` | Suffix (e.g., " XP") |
| `variant` | `"float" \| "pop" \| "slide"` | `"float"` | Animation style |
| `color` | `"default" \| "green" \| "gold" \| "blue"` | `"default"` | Text color |
| `duration` | `number` | `1000` | Animation duration in ms |
| `onAnimationEnd` | `() => void` | - | Callback when animation ends |

### usePointsAnimation Hook

```tsx
const { items, addPoints, removeItem } = usePointsAnimation()
```

Returns:
- `items` - Array of animation items
- `addPoints(value, options?)` - Add a new animation
- `removeItem(id)` - Remove an animation

## Accessibility

- Animations respect `prefers-reduced-motion`
- Points values announced to screen readers
