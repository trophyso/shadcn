## Usage

```tsx
import { Leaderboard } from "@/components/ui/leaderboard"
```

```tsx
<Leaderboard
  rankings={[
    { userId: "1", userName: "Alice", rank: 1, value: 2500 },
    { userId: "2", userName: "Bob", rank: 2, value: 2200 },
    { userId: "3", userName: "Charlie", rank: 3, value: 1800 },
  ]}
/>
```

## Examples

### Variants

```tsx
// Table style with header (default)
<Leaderboard rankings={rankings} variant="table" />

// Card style - each entry in a separate card
<Leaderboard rankings={rankings} variant="cards" />

// Compact style - smaller text, tighter spacing
<Leaderboard rankings={rankings} variant="compact" />
```

### Highlight Current User

```tsx
<Leaderboard
  rankings={rankings}
  currentUserId="user_123"
/>
// Current user row is highlighted with "(you)" label
```

### Limit Displayed Entries

```tsx
<Leaderboard
  rankings={rankings}
  limit={10}
/>
// Shows top 10 with "+X more" indicator if there are more
```

### Custom Value Label

```tsx
// Points
<Leaderboard rankings={rankings} valueLabel="Points" />

// XP
<Leaderboard rankings={rankings} valueLabel="XP" />

// Tasks
<Leaderboard rankings={rankings} valueLabel="Tasks" />
```

### Hide Value Column

```tsx
<Leaderboard rankings={rankings} showValue={false} />
```

### Without Avatars

```tsx
<Leaderboard rankings={rankings} showAvatar={false} />
```

### Clickable Rows

```tsx
<Leaderboard
  rankings={rankings}
  onUserClick={(ranking) => {
    router.push(`/profile/${ranking.userId}`)
  }}
/>
```

### Custom Empty State

```tsx
<Leaderboard
  rankings={[]}
  emptyState={
    <div className="text-center py-8">
      <p className="text-lg font-medium">Be the first!</p>
      <p className="text-muted-foreground">Complete tasks to appear on the leaderboard</p>
    </div>
  }
/>
```

## API Reference

### Leaderboard

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `leaderboard` | `LeaderboardResponse` | - | Full leaderboard data object |
| `rankings` | `LeaderboardRanking[]` | - | Rankings array (alternative to leaderboard prop) |
| `variant` | `"table" \| "cards" \| "compact"` | `"table"` | Layout variant |
| `currentUserId` | `string` | - | Highlight this user's row |
| `limit` | `number` | - | Max entries to display |
| `showValue` | `boolean` | `true` | Show score/value column |
| `valueLabel` | `string` | `"Score"` | Label for value column |
| `showAvatar` | `boolean` | `true` | Show avatar placeholders |
| `emptyState` | `ReactNode` | - | Custom empty state |
| `onUserClick` | `(ranking) => void` | - | Row click handler |

### LeaderboardRanking

```ts
interface LeaderboardRanking {
  userId: string
  userName: string | null
  rank: number
  value: number
}
```

### LeaderboardResponse

```ts
interface LeaderboardResponse {
  rankings: LeaderboardRanking[]
}
```
