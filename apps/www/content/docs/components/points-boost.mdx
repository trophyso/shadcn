## Usage

Import the component:

```tsx
import { PointsBoost } from "@/components/ui/points-boost"
```

```tsx
<PointsBoost
  boost={{
    name: "Double Points Weekend",
    description: "Earn extra points for all tracked actions",
    multiplier: 2,
    cta: {
      link: "#",
      text: "Activate boost",
    },
  }}
/>
```

```tsx
<PointsBoost
  boost={{
    name: "Limited-Time Points Boost",
    description: "Boost expires soon, activate now to maximize your points",
    multiplier: 3,
    cta: {
      link: "#",
      text: "Activate now",
    },
    endDate: new Date(Date.now() + 1000 * 60 * 60 * 11 + 1000 * 60 * 29 + 1000 * 36).toISOString(),
  }}
/>
```

## With Trophy

Use the [Trophy SDK](https://docs.trophy.so/api-reference/client-libraries) to fetch boosts for a points system server-side:

<OpenApiCodeSamples spec="application.yml" method="get" endpoint="/points/{key}/boosts" />

Then pass and transform the response into `PointsBoost` props in your UI layer:

```tsx
const activeOrScheduledBoost = (response.boosts ?? [])[0]

const boost = activeOrScheduledBoost && {
  name: activeOrScheduledBoost.name,
  description: activeOrScheduledBoost.description,
  multiplier: activeOrScheduledBoost.multiplier,
  endDate: activeOrScheduledBoost.endDate,
  cta: {
    link: "/rewards",
    text: "View boost",
  },
}

{boost ? <PointsBoost boost={boost} /> : null}
```
