## Usage

```tsx
import { LeaderboardPodium } from "@/components/ui/leaderboard-podium"
```

```tsx
<LeaderboardPodium
  rankings={[
    { userId: "1", userName: "Alice", rank: 1, value: 2500 },
    { userId: "2", userName: "Bob", rank: 2, value: 2200 },
    { userId: "3", userName: "Charlie", rank: 3, value: 1800 },
  ]}
/>
```

## Examples

### Sizes

```tsx
<LeaderboardPodium rankings={rankings} size="sm" />
<LeaderboardPodium rankings={rankings} size="default" />
<LeaderboardPodium rankings={rankings} size="lg" />
```

### Medal Styles

```tsx
// Classic - medal badge overlay on avatar
<LeaderboardPodium rankings={rankings} medalStyle="classic" />

// Modern - rounded corners on podium blocks
<LeaderboardPodium rankings={rankings} medalStyle="modern" />

// Minimal - no medal badge, just colored avatar ring
<LeaderboardPodium rankings={rankings} medalStyle="minimal" />
```

### Without Values

```tsx
<LeaderboardPodium rankings={rankings} showValue={false} />
```

### Without Avatars

Shows medal icons instead of avatar initials:

```tsx
<LeaderboardPodium rankings={rankings} showAvatar={false} />
```

### Hero Section

```tsx
<div className="bg-gradient-to-b from-purple-900 to-purple-950 py-12 px-4 rounded-xl">
  <h2 className="text-2xl font-bold text-white text-center mb-8">
    This Week's Champions
  </h2>
  <LeaderboardPodium
    rankings={rankings}
    size="lg"
    medalStyle="modern"
  />
</div>
```

### With Full Leaderboard Below


### Partial Podium

Works with fewer than 3 users:

```tsx
// Only 2 users - shows 2nd and 1st place
<LeaderboardPodium
  rankings={[
    { userId: "1", userName: "Alice", rank: 1, value: 2500 },
    { userId: "2", userName: "Bob", rank: 2, value: 2200 },
  ]}
/>

// Only 1 user - shows just 1st place
<LeaderboardPodium
  rankings={[
    { userId: "1", userName: "Alice", rank: 1, value: 2500 },
  ]}
/>
```

## API Reference

### LeaderboardPodium

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `rankings` | `LeaderboardRanking[]` | Required | Top rankings (uses first 3) |
| `size` | `"sm" \| "default" \| "lg"` | `"default"` | Podium size |
| `showValue` | `boolean` | `true` | Show score below name |
| `showAvatar` | `boolean` | `true` | Show avatar initials |
| `medalStyle` | `"classic" \| "modern" \| "minimal"` | `"classic"` | Medal display style |

### LeaderboardRanking

```ts
interface LeaderboardRanking {
  userId: string
  userName: string | null
  rank: number
  value: number
}
```
